# تقرير شامل: تطبيق توقع أداء الطلاب بلغة C#

## نظرة عامة على المشروع

تم تطوير تطبيق Windows Forms احترافي باستخدام لغة C# يهدف إلى توقع الأداء الأكاديمي للطلاب باستخدام خوارزمية الانحدار الخطي المدربة مسبقاً على بيانات 10,000 طالب.

    {
        MessageBox.Show("يرجى إدخال اسم المستخدم وكلمة المرور", "خطأ");

public static double PredictPerformance(int hoursStudied, int previousScores, 
    bool hasExtracurricular, int sleepHours, int samplePapers)
{
    int extracurricularEncoded = hasExtracurricular ? 1 : 0;

    double prediction = Intercept +
        (Coefficients[0] * hoursStudied) +
        (Coefficients[1] * previousScores) +
        (Coefficients[2] * extracurricularEncoded) +
        (Coefficients[3] * sleepHours) +
        (Coefficients[4] * samplePapers);

    return Math.Max(0, Math.Min(100, prediction));
}
```

- **الدرجات السابقة**: كل نقطة إضافية تحسن الأداء بـ 1.02 نقطة

public static string GetRecommendations(int hoursStudied, int previousScores, 
    bool hasExtracurricular, int sleepHours, int samplePapers)
{
    var recommendations = new StringBuilder();
    
    if (hoursStudied < 6)
        recommendations.AppendLine("• زيادة ساعات الدراسة");
    
    if (previousScores < 70)
        recommendations.AppendLine("• التركيز على تحسين الدرجات الحالية");
    
    if (!hasExtracurricular)
        recommendations.AppendLine("• المشاركة في الأنشطة اللامنهجية");
    
    if (sleepHours < 7)
        recommendations.AppendLine("• زيادة ساعات النوم");
    
    return recommendations.ToString();
}
```
